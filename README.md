jquery-slideshow
================

slideshow

Usage:

<!-- in html -->
<div id="slideshow">
    <div class="slides">
        <div class="slide"><!-- 1st slide -->
            <!-- contents here -->
        </div>
        <div class="slide"><!-- 2nd slide -->
            <!-- contents here -->
        </div>
        <div class="slide"><!-- 3rd slide -->
            <!-- contents here -->
        </div>
        <!-- more slides -->
    </div>
    <div class="buttons"><!-- buttons div are optional -->
        <span class="button"></span>
        <span class="button"></span>
        <span class="button"></span>
    </div>
</div>

<!-- in script -->
$("#slideshow").slideshow();
